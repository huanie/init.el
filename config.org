  (defun my/eval-buffer ()
    "Execute the current buffer as Lisp code.
  Need this because defcustom, defvar etc
  will not be evaluated with default `eval-buffer'."
    (interactive)
    (save-excursion
      (goto-char (point-min))
      (while (not (eobp))
	(forward-sexp)
	(eval-last-sexp nil))))
